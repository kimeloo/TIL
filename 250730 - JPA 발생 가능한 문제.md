// 문제가 되는 코드
List<Live> lives = liveRepository.findAll();
for (Live live : lives) {
    System.out.println(live.getAuctions().size()); // N+1 발생!
}

// 해결 방법 1: Fetch Join
@Query("SELECT l FROM Live l JOIN FETCH l.auctions WHERE l.isDeleted = false")
